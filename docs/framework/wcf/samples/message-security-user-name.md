```xml  
        </security>  
      </binding>

        <serviceCredentials>  
          <serviceCertificate findValue="localhost" storeLocation="LocalMachine" storeName="My" x509FindType="FindBySubjectName" />

    </serviceBehaviors>  
  </behaviors>

```
