```csharp
public interface ICalculator

    [OperationContract]
```

<span data-ttu-id="77b7d-114">Hizmet uygulaması, hesaplar ve uygun sonucunu döndürür:</span><span class="sxs-lookup"><span data-stu-id="77b7d-114">The service implementation calculates and returns the appropriate result:</span></span>

```csharp
{
    public double Add(double n1, double n2)
    {
        return n1 + n2;
    }
    public double Subtract(double n1, double n2)
    {
        return n1 - n2;
    }
    public double Multiply(double n1, double n2)
    {

    }

```

```xml
    </bindings>

    <!--For debugging purposes set the includeExceptionDetailInFaults attribute to true-->

      <serviceBehaviors>
```
