    public SyntaxTree Syntax { get; }

}  
```

 <span data-ttu-id="a3a2c-288">**Örneğin 6 Düzelt**</span><span class="sxs-lookup"><span data-stu-id="a3a2c-288">**Fix for example 6**</span></span>  
  
 <span data-ttu-id="a3a2c-289">Tamamlanan kaldırmak için <xref:System.Threading.Tasks.Task> ayırma, görev nesnesi tamamlanmış sonucuyla önbelleğe alabilir:</span><span class="sxs-lookup"><span data-stu-id="a3a2c-289">To remove the completed <xref:System.Threading.Tasks.Task> allocation, you can cache the Task object with the completed result:</span></span>

    private Task<SyntaxTree> cachedResult;  
  
    public Task<SyntaxTree> GetSyntaxTreeAsync()  
    {  
        return this.cachedResult ??   
               (this.cachedResult = GetSyntaxTreeUncachedAsync());

    private async Task<SyntaxTree> GetSyntaxTreeUncachedAsync()

}  
```

data-ttu-id="a3a2c-295">Artık, önbelleğe alınan sonucu olan bir <xref:System.Threading.Tasks.Task>var.
