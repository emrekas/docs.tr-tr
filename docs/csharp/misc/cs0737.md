{  
    // Default access of private with no modifier.  
    int Return42();  
    // Try the following line instead.

    int Return42() { return (42); }

{  
    public static int Main(string[] args)

        Struct1 s1 = new Struct1();  
  
        return (1);  
    }  
  
}
