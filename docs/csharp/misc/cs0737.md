## <a name="example"></a><span data-ttu-id="ee9e7-109">Örnek</span><span class="sxs-lookup"><span data-stu-id="ee9e7-109">Example</span></span>

{  
    // Default access of private with no modifier.  
    int Return42();  
    // Try the following line instead.  
    // public int Return42();

    int Return42() { return (42); }  
}

{  
    public static int Main(string[] args)  
    {  
        Struct1 s1 = new Struct1();  
  
        return (1);  
    }  
  
}
