## <a name="example"></a><span data-ttu-id="8c831-106">Örnek</span><span class="sxs-lookup"><span data-stu-id="8c831-106">Example</span></span>  
 <span data-ttu-id="8c831-107">Aşağıdaki örnek CS0643 oluşturur:</span><span class="sxs-lookup"><span data-stu-id="8c831-107">The following sample generates CS0643:</span></span>

```csharp  
// CS0643.cs  
using System;  
using System.Runtime.InteropServices;  
  
[AttributeUsage(AttributeTargets.Class)]  
public class MyAttribute : Attribute  
{  
    public MyAttribute()  
    {  
    }  
  
    public int x;  
}  
  
[MyAttribute(x = 5, x = 6)]   // CS0643, error setting x twice  
// try the following line instead  
// [MyAttribute(x = 5)]

{  
}

{  
    public static void Main ()  
    {  
    }  
}  
```
